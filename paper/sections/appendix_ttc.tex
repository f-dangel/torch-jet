    \gamma_{\vk \vl} = \sum_{0 < \vm \leq \vk} (-1)^{|\vk - \vm|} 
    \left(
    \begin{matrix}
        \vk \\
        \vm
    \end{matrix}
    \right)

        |\vk|\frac{\vm}{|\vm|} \\

    \frac{|\vm|}{|\vk|}
    \right)^{|\vk|}.
\end{equation}
